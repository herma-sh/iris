## Kitty Graphics Protocol

### Sequence Format

```
ESC _ G <control-key>=<value>:<control-key>=<value>:... ; <base64-data> ESC \
```

### Control Keys

| Key | Meaning |
|-----|---------|
| `a` | Action: T=transmit, d=delete, q=query, p=place |
| `f` | Format: 24=RGB, 32=RGBA, 100=PNG |
| `t` | Transmission: a=auto, d=direct, s=stream |
| `s` | Width in pixels |
| `v` | Height in pixels |
| `c` | Columns (cell placement) |
| `r` | Rows (cell placement) |
| `i` | Image ID |
| `o` | Compression |

use iris_core::grid::GridSize;

/// Inline image in terminal.
#[derive(Clone, Debug)]
pub struct InlineImage {
    /// Unique image ID.
    pub id: u32,
    /// Image data (RGBA).
    pub data: Vec<u8>,
    /// Width in pixels.
    pub width: u32,
    /// Height in pixels.
    pub height: u32,
    /// Placement in cells.
    pub placement: ImagePlacement,
    /// Z-index (layering).
    pub z_index: i32,
}

#[derive(Clone, Debug)]
pub struct ImagePlacement {
    /// Column position.
    pub col: u32,
    /// Row position.
    pub row: u32,
    /// X offset within cell.
    pub x_offset: u32,
    /// Y offset within cell.
    pub y_offset: u32,
    /// Width in cells.
    pub cols: u32,
    /// Height in cells.
    pub rows: u32,
}

/// Kitty graphics handler.
pub struct KittyGraphics {
    /// Active images.
    images: HashMap<u32, InlineImage>,
    /// Next image ID.
    next_id: u32,
    /// Texture cache.
    textures: HashMap<u32, wgpu::Texture>,
}

impl KittyGraphics {
    pub fn new() -> Self {
        Self {
            images: HashMap::new(),
            next_id: 1,
            textures: HashMap::new(),
        }
    }
    
    /// Process Kitty APC sequence.
    pub fn process(&mut self, data: &str) -> Result<Option<InlineImage>> {
        let (control, payload) = Self::parse_apc(data)?;
        let params = Self::parse_control(&control)?;
        
        let action = params.get("a").map(|s| s.as_str()).unwrap_or("T");
        
        match action {
            "T" => self.handle_transmit(&params, payload),
            "d" => self.handle_delete(&params),
            "q" => self.handle_query(&params),
            "p" => self.handle_place(&params),
            _ => Ok(None),
        }
    }
    
    fn handle_transmit(&mut self, params: &HashMap<String, String>, payload: &[u8]) -> Result<Option<InlineImage>> {
        let format = params.get("f").map(|s| s.as_str()).unwrap_or("100");
        let width = params.get("s").and_then(|s| s.parse().ok()).unwrap_or(0);
        let height = params.get("v").and_then(|s| s.parse().ok()).unwrap_or(0);
        let cols = params.get("c").and_then(|s| s.parse().ok()).unwrap_or(1);
        let rows = params.get("r").and_then(|s| s.parse().ok()).unwrap_or(1);
        
        // Decode image
        let (data, width, height) = if format == "100" {
            // PNG
            Self::decode_png(payload)?
        } else {
            // Raw RGB/RGBA
            let data = if params.contains_key("o") {
                Self::decompress(payload)?
            } else {
                payload.to_vec()
            };
            (data, width, height)
        };
        
        let id = params.get("i").and_then(|s| s.parse().ok()).unwrap_or_else(|| {
            let id = self.next_id;
            self.next_id += 1;
            id
        });
        
        let image = InlineImage {
            id,
            data,
            width,
            height,
            placement: ImagePlacement {
                col: 0,
                row: 0,
                x_offset: 0,
                y_offset: 0,
                cols,
                rows,
            },
            z_index: 0,
        };
        
        self.images.insert(id, image.clone());
        Ok(Some(image))
    }
    
    fn handle_delete(&mut self, params: &HashMap<String, String>) -> Result<Option<InlineImage>> {
        if let Some(id) = params.get("i").and_then(|s| s.parse().ok()) {
            self.images.remove(&id);
        }
        Ok(None)
    }
}
