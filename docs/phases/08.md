## OSC 133 Protocol

| Sequence | Name | Meaning |
|----------|------|---------|
| `OSC 133 ; A ST` | MarkPromptStart | Command line start |
| `OSC 133 ; B ST` | MarkPromptEnd | Command end, output begins |
| `OSC 133 ; C ST` | MarkOutputStart | Output begins |
| `OSC 133 ; D ; %d ST` | MarkOutputEnd | Output ends, exit code |

## API Contracts

### shell_integration/mod.rs

```rust
use iris_core::cursor::CursorPosition;

/// Shell integration state.
pub struct ShellIntegration {
    /// Shell type.
    shell: ShellType,
    /// Prompt markers.
    prompts: Vec<PromptMarker>,
    /// Current prompt state.
    state: PromptState,
}

#[derive(Clone, Debug)]
pub enum ShellType {
    Bash,
    Zsh,
    Fish,
    PowerShell,
}

#[derive(Clone, Debug)]
pub struct PromptMarker {
    /// Line position.
    pub position: CursorPosition,
    /// Text of the command.
    pub command: Option<String>,
    /// Output start position.
    pub output_start: Option<CursorPosition>,
    /// Output end position.
    pub output_end: Option<CursorPosition>,
    /// Exit code.
    pub exit_code: Option<i32>,
    /// Timestamp.
    pub timestamp: std::time::Instant,
}

#[derive(Clone, Copy, Debug, PartialEq)]
pub enum PromptState {
    Normal,
    PromptStart,
    PromptEnd,
    OutputStart,
}

impl ShellIntegration {
    pub fn new() -> Self {
        Self {
            shell: ShellType::detect(),
            prompts: Vec::new(),
            state: PromptState::Normal,
        }
    }
    
    /// Handle OSC 133 sequence.
    pub fn handle_osc133(&mut self, params: &str, grid: &mut iris_core::grid::Grid) {
        match params {
            "A" => self.handle_prompt_start(grid),
            "B" => self.handle_prompt_end(grid),
            "C" => self.handle_output_start(grid),
            _ if params.starts_with("D;") => self.handle_output_end(params, grid),
            _ => {}
        }
    }
    
    fn handle_prompt_start(&mut self, grid: &iris_core::grid::Grid) {
        let marker = PromptMarker {
            position: Self::get_cursor_position(grid),
            command: None,
            output_start: None,
            output_end: None,
            exit_code: None,
            timestamp: std::time::Instant::now(),
        };
        self.prompts.push(marker);
        self.state = PromptState::PromptStart;
    }
    
    fn handle_prompt_end(&mut self, grid: &iris_core::grid::Grid) {
        if let Some(marker) = self.prompts.last_mut() {
            // Extract command text
            marker.command = Some(Self::extract_command(grid, marker.position));
        }
        self.state = PromptState::PromptEnd;
    }
    
    /// Get previous prompt.
    pub fn previous_prompt(&self, current: CursorPosition) -> Option<&PromptMarker> {
        self.prompts.iter()
            .rev()
            .find(|m| m.position.row < current.row)
    }
    
    /// Get next prompt.
    pub fn next_prompt(&self, current: CursorPosition) -> Option<&PromptMarker> {
        self.prompts.iter()
            .find(|m| m.position.row > current.row)
    }
    
    /// Get last prompt.
    pub fn last_prompt(&self) -> Option<&PromptMarker> {
        self.prompts.last()
    }
    
    /// Get last command output.
    pub fn get_last_output(&self, grid: &iris_core::grid::Grid) -> Option<String> {
        let marker = self.prompts.last()?;
        let start = marker.output_start?;
        let end = marker.output_end?;
        
        let mut text = String::new();
        for row in start.row..=end.row {
            if let Some(line) = grid.row(row) {
                for cell in line {
                    text.push(cell.char);
                }
                text.push('\n');
            }
        }
        Some(text)
    }
}

impl ShellType {
    /// Detect shell from $SHELL.
    pub fn detect() -> Self {
        std::env::var("SHELL")
            .map(|s| match s.as_str() {
                s if s.contains("bash") => ShellType::Bash,
                s if s.contains("zsh") => ShellType::Zsh,
                s if s.contains("fish") => ShellType::Fish,
                s if s.contains("pwsh") || s.contains("powershell") => ShellType::PowerShell,
                _ => ShellType::Bash,
            })
            .unwrap_or(ShellType::Bash)
    }
    
    /// Get integration script.
    pub fn integration_script(&self) -> &'static str {
        match self {
            ShellType::Bash => include_str!("../scripts/bash_integration.sh"),
            ShellType::Zsh => include_str!("../scripts/zsh_integration.zsh"),
            ShellType::Fish => include_str!("../scripts/fish_integration.fish"),
            ShellType::PowerShell => include_str!("../scripts/pwsh_integration.ps1"),
        }
    }
}
```

### scripts/bash_integration.sh

```bash
# Iris terminal integration for Bash

__iris_prompt_start() {
    printf '\e]133;A\e\\'
}

__iris_prompt_end() {
    printf '\e]133;B\e\\'
}

__iris_output_start() {
    printf '\e]133;C\e\\'
}

__iris_command_executed() {
    __iris_output_start
    __iris_save_exit_code=$?
}

__iris_output_end() {
    printf '\e]133;D;%s\e\\' "$__iris_save_exit_code"
}

# Add to PROMPT_COMMAND
if [[ -z "$PROMPT_COMMAND" ]]; then
    PROMPT_COMMAND='__iris_prompt_start'
else
    PROMPT_COMMAND="__iris_prompt_start; $PROMPT_COMMAND"
fi

# Mark prompt end
PS1="\[\e]133;B\e\\\]$PS1"

# Pre-exec hook
trap '__iris_command_executed' DEBUG
```
