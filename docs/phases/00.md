use std::fmt;

pub type Result<T> = std::result::Result<T, Error>;

#[derive(Debug)]
pub enum Error {
    /// Invalid cell position
    InvalidPosition { row: usize, col: usize },
    /// Grid operation failed
    GridError(String),
    /// Invalid mode transition
    InvalidModeTransition { from: String, to: String },
    /// Resize failed
    ResizeFailed { reason: String },
    /// Buffer overflow
    BufferOverflow { capacity: usize, requested: usize },
}

impl fmt::Display for Error {
    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
        match self {
            Self::InvalidPosition { row, col } => {
                write!(f, "Invalid position: ({}, {})", row, col)
            }
            Self::GridError(msg) => write!(f, "Grid error: {}", msg),
            Self::InvalidModeTransition { from, to } => {
                write!(f, "Invalid mode transition: {} -> {}", from, to)
            }
            Self::ResizeFailed { reason } => write!(f, "Resize failed: {}", reason),
            Self::BufferOverflow { capacity, requested } => {
                write!(f, "Buffer overflow: capacity {}, requested {}", capacity, requested)
            }
        }
    }
}

impl std::error::Error for Error {}

/// A single cell in the terminal grid.
#[derive(Clone, Debug, PartialEq)]
pub struct Cell {
    /// The character stored in this cell.
    pub char: char,
    /// The width of this cell (1 or2 for wide characters).
    pub width: CellWidth,
    /// Attribute flags (bold, italic, etc.).
    pub attrs: CellAttrs,
}

impl Default for Cell {
    fn default() -> Self {
        Self {
            char: ' ',
            width: CellWidth::Single,
            attrs: CellAttrs::default(),
        }
    }
}

impl Cell {
    /// Create a new cell with the given character.
    pub fn new(char: char) -> Self {
        let width = CellWidth::from_char(char);
        Self {
            char,
            width,
            attrs: CellAttrs::default(),
        }
    }

    /// Create a cell with attributes.
    pub fn with_attrs(char: char, attrs: CellAttrs) -> Self {
        let width = CellWidth::from_char(char);
        Self { char, width, attrs }
    }

    /// Check if this is a wide character cell (CJK, emoji, etc.)
    pub fn is_wide(&self) -> bool {
        self.width == CellWidth::Double
    }

    /// Check if this cell is empty (space).
    pub fn is_empty(&self) -> bool {
        self.char == ' '
    }

    /// Reset the cell to default state.
    pub fn reset(&mut self) {
        self.char = ' ';
        self.width = CellWidth::Single;
        self.attrs = CellAttrs::default();
    }
}

/// Grid size specification.
#[derive(Clone, Copy, Debug, PartialEq, Eq)]
pub struct GridSize {
    pub rows: usize,
    pub cols: usize,
}

impl Default for GridSize {
    fn default() -> Self {
        Self {rows: 24, cols: 80 }
    }
}

/// The terminal grid - a 2D array of cells.
pub struct Grid {
    /// Grid size.
    size: GridSize,
    /// Cell storage (row-major order).
    cells: Vec<Vec<Cell>>,
    /// Damage tracker for efficient redraws.
    damage: DamageTracker,
}

impl Grid {
    /// Create a new grid with the given size.
    pub fn new(size: GridSize) -> Self {
        let cells = (0..size.rows)
            .map(|_| (0..size.cols).map(|_| Cell::default()).collect())
            .collect();
        
        Self {
            size,
            cells,
            damage: DamageTracker::new(),
        }
    }

    /// Get grid size.
    pub fn size(&self) -> GridSize {
        self.size
    }

    /// Get the number of rows.
    pub fn rows(&self) -> usize {
        self.size.rows
    }

    /// Get the number of columns.
    pub fn cols(&self) -> usize {
        self.size.cols
    }

    /// Get a cell at the given position.
    pub fn cell(&self, row: usize, col: usize) -> Option<&Cell> {
        self.cells.get(row)?.get(col)
    }

    /// Get a mutable cell at the given position.
    pub fn cell_mut(&mut self, row: usize, col: usize) -> Option<&mut Cell> {
        self.cells.get_mut(row)?.get_mut(col)
    }

    /// Write a character at the given position.
    pub fn write(&mut self, row: usize, col: usize, cell: Cell) -> Result<()> {
        if row >= self.size.rows || col >= self.size.cols {
            return Err(Error::InvalidPosition { row, col });
        }
        
        self.cells[row][col] = cell;
        self.damage.mark(row, col);
        
        // If wide character, mark next cell as continuation.
        if cell.is_wide() && col + 1 < self.size.cols {
            let mut cont = Cell::default();
            cont.attrs = cell.attrs.clone();
            self.cells[row][col + 1] = cont;
            self.damage.mark(row, col + 1);
        }
        
        Ok(())
    }

    /// Clear a row.
    pub fn clear_row(&mut self, row: usize) -> Result<()> {
        if row >= self.size.rows {
            return Err(Error::InvalidPosition { row, col: 0 });
        }
        
        for cell in &mut self.cells[row] {
            cell.reset();
        }
        self.damage.mark_row(row);
        
        Ok(())
    }

    /// Clear the entire grid.
    pub fn clear(&mut self) {
        for row in 0..self.size.rows {
            for cell in &mut self.cells[row] {
                cell.reset();
            }
        }
        self.damage.mark_all();
    }

    /// Scroll the grid up by n lines.
    pub fn scroll_up(&mut self, n: usize) {
        // Remove top n rows
        for _ in 0..n {
            self.cells.remove(0);
        }
        
        // Add n new rows at bottom
        for _ in 0..n {
            let row = (0..self.size.cols).map(|_| Cell::default()).collect();
            self.cells.push(row);
        }
        
        self.damage.mark_all();
    }

    /// Scroll the grid down by n lines.
    pub fn scroll_down(&mut self, n: usize) {
        // Remove bottom n rows
        for _ in 0..n {
            self.cells.pop();
        }
        
        // Insert n new rows at top
        for _ in 0..n {
            let row = (0..self.size.cols).map(|_| Cell::default()).collect();
            self.cells.insert(0, row);
        }
        
        self.damage.mark_all();
    }

    /// Resize the grid.
    pub fn resize(&mut self, new_size: GridSize) -> Result<()> {
        // Resize rows
        self.cells.truncate(new_size.rows);
        while self.cells.len() < new_size.rows {
            let row = (0..new_size.cols).map(|_| Cell::default()).collect();
            self.cells.push(row);
        }
        
        // Resize columns in each row
        for row in &mut self.cells {
            row.truncate(new_size.cols);
            while row.len() < new_size.cols {
                row.push(Cell::default());
            }
        }
        
        self.size = new_size;
        self.damage.mark_all();
        
        Ok(())
    }

    /// Get damaged regions since last take.
    pub fn take_damage(&mut self) -> Vec<DamageRegion> {
        self.damage.take()
    }

    /// Get a row as a slice.
    pub fn row(&self, row: usize) -> Option<&[Cell]> {
        self.cells.get(row).map(|r| r.as_slice())
    }
}

impl Index<usize> for Grid {
    type Output = [Cell];
    
    fn index(&self, row: usize) -> &Self::Output {
        &self.cells[row]
    }
}

impl IndexMut<usize> for Grid {
    fn index_mut(&mut self, row: usize) -> &mut Self::Output {
        &mut self.cells[row]
    }
}

use std::collections::HashSet;

/// A region of the grid that has been modified.
#[derive(Clone, Debug, PartialEq, Eq)]
pub struct DamageRegion {
    pub start_row: usize,
    pub end_row: usize,
    pub start_col: usize,
    pub end_col: usize,
}

impl DamageRegion {
    pub fn new(start_row: usize, end_row: usize, start_col: usize, end_col: usize) -> Self {
        Self { start_row, end_row, start_col, end_col }
    }

    pub fn single(row: usize, col: usize) -> Self {
        Self::new(row, row, col, col)
    }

    pub fn row(row: usize) -> Self {
        Self::new(row, row, 0, usize::MAX)
    }

    pub fn all() -> Self {
        Self::new(0, usize::MAX, 0, usize::MAX)
    }
}

/// Tracks damage regions for efficient rendering.
pub struct DamageTracker {
    /// Set of damaged positions.
    damaged: HashSet<(usize, usize)>,
    /// Whether the entire grid is damaged.
    all_damaged: bool,
}

impl DamageTracker {
    pub fn new() -> Self {
        Self {
            damaged: HashSet::new(),
            all_damaged: false,
        }
    }

    /// Mark a single cell as damaged.
    pub fn mark(&mut self, row: usize, col: usize) {
        self.damaged.insert((row, col));
    }

    /// Mark an entire row as damaged.
    pub fn mark_row(&mut self, row: usize) {
        // Use a sentinel to indicate full row
        self.damaged.insert((row, usize::MAX));
    }

    /// Mark the entire grid as damaged.
    pub fn mark_all(&mut self) {
        self.all_damaged = true;
        self.damaged.clear();
    }

    /// Take and return all damaged regions, clearing the tracker.
    pub fn take(&mut self) -> Vec<DamageRegion> {
        if self.all_damaged {
            self.all_damaged = false;
            return vec![DamageRegion::all()];
        }

        if self.damaged.is_empty() {
            return vec![];
        }

        // Group damaged cells into regions
        let damaged = std::mem::take(&mut self.damaged);
        let mut regions = Vec::new();
        
        // For simplicity, return individual positions
        // A more sophisticated implementation would merge adjacent cells
        for (row, col) in damaged {
            if col == usize::MAX {
                regions.push(DamageRegion::row(row));
            } else {
                regions.push(DamageRegion::single(row, col));
            }
        }

        regions
    }

    /// Check if there are any damaged regions.
    pub fn is_damaged(&self) -> bool {
        self.all_damaged || !self.damaged.is_empty()
    }

    /// Check if a specific cell is damaged.
    pub fn is_cell_damaged(&self, row: usize, col: usize) -> bool {
        self.all_damaged || self.damaged.contains(&(row, col))
    }
}

impl Default for DamageTracker {
    fn default() -> Self {
        Self::new()
    }
}

            visible: true,
            blink: true,
            fg: Color::Default,
            bg: Color::Default,

impl Cursor {
    /// Create a new cursor at the origin.
    pub fn new() -> Self {
        Self::default()
    }

    /// Move the cursor to a position.
    pub fn move_to(&mut self, row: usize, col: usize) {
        self.position.row = row;
        self.position.col = col;
    }

    /// Move the cursor up by n rows.
    pub fn move_up(&mut self, n: usize) {
        self.position.row = self.position.row.saturating_sub(n);
    }

    /// Move the cursor down by n rows.
    pub fn move_down(&mut self, n: usize, max_rows: usize) {
        self.position.row = (self.position.row + n).min(max_rows.saturating_sub(1));
    }

    /// Move the cursor left by n columns.
    pub fn move_left(&mut self, n: usize) {
        self.position.col = self.position.col.saturating_sub(n);
    }

    /// Move the cursor right by n columns.
    pub fn move_right(&mut self, n: usize, max_cols: usize) {
        self.position.col = (self.position.col + n).min(max_cols.saturating_sub(1));
    }

    /// Save the cursor position.
    pub fn save(&self) -> SavedCursor {
        SavedCursor {
            position: self.position,
            style: self.style,
        }
    }

    /// Restore the cursor position.
    pub fn restore(&mut self, saved: &SavedCursor) {
        self.position = saved.position;
        self.style = saved.style;
    }
}

            4 => Some(Self::Insert),

            25 => Some(Self::CursorVisible),
            12 => Some(Self::CursorBlink),
            1049 => Some(Self::AlternateScreen),
            2004 => Some(Self::BracketedPaste),
            1004 => Some(Self::FocusEvent),
            2026 => Some(Self::SynchronizedOutput),
            _ => None,
        }
    }

    pub fn to_decset(&self) -> u16 {
        match self {
            Self::Origin => 6,
            Self::Wrap => 7,
            Self::Insert => 4,
            Self::Newline => 20,
            Self::CursorVisible => 25,

            Self::AlternateScreen => 1049,
            Self::BracketedPaste => 2004,
            Self::FocusEvent => 1004,
            Self::SynchronizedOutput => 2026,
        }
    }

use crate::cell::Cell;
use crate::cursor::{Cursor, CursorPosition};
use crate::error::Result;
use crate::grid::{Grid, GridSize};
use crate::modes::TerminalModes;

/// The main terminal state.
pub struct Terminal {
    /// The grid (visible buffer).
    pub grid: Grid,
    /// The cursor.
    pub cursor: Cursor,
    /// Terminal modes.
    pub modes: TerminalModes,
    /// Scroll region (top, bottom).
    pub scroll_region: (usize, usize),
    /// Saved cursor state.
    saved_cursor: Option<CursorPosition>,
    /// Saved cursor position (DECSC).
    decsc_cursor: Option<CursorPosition>,
    /// Saved cursor for alternate buffer.
    alternate_cursor: Option<Cursor>,
    /// Alternate buffer (if active).
    alternate_grid: Option<Grid>,
}

impl Terminal {
    /// Create a new terminal with the given size.
    pub fn new(rows: usize, cols: usize) -> Self {
        let grid = Grid::new(GridSize { rows, cols });
        let scroll_region = (0, rows);
        
        Self {
            grid,
            cursor: Cursor::new(),
            modes: TerminalModes::new(),
            scroll_region,
            saved_cursor: None,
            decsc_cursor: None,
            alternate_cursor: None,
            alternate_grid: None,
        }
    }

    /// Get the current cursor position.
    pub fn position(&self) -> CursorPosition {
        self.cursor.position
    }

    /// Write a character at the cursor position.
    pub fn write_char(&mut self, c: char) -> Result<()> {
        let pos = self.cursor.position;
        let cell = Cell::new(c);
        self.grid.write(pos.row, pos.col, cell)?;
        
        // Move cursor
        if pos.col+ cell.width.columns() < self.grid.cols() {
            self.cursor.position.col += cell.width.columns();
        } else if self.modes.wrap {
            self.cursor.position.col = 0;
            self.cursor.position.row += 1;
        }
        
        Ok(())
    }

    /// Move cursor to position.
    pub fn move_cursor(&mut self, row: usize, col: usize) {
        let row = row.min(self.grid.rows().saturating_sub(1));
        let col = col.min(self.grid.cols().saturating_sub(1));
        self.cursor.move_to(row, col);
    }

    /// Move cursor relative to current position.
    pub fn move_cursor_relative(&mut self, row_delta: isize, col_delta: isize) {
        let new_row = if row_delta >= 0 {
            (self.cursor.position.row + row_delta as usize)
                .min(self.grid.rows().saturating_sub(1))
        } else {
            self.cursor.position.row.saturating_sub((-row_delta) as usize)
        };
        
        let new_col = if col_delta >= 0 {
            (self.cursor.position.col + col_delta as usize)
                .min(self.grid.cols().saturating_sub(1))
        } else {
            self.cursor.position.col.saturating_sub((-col_delta) as usize)
        };
        
        self.cursor.move_to(new_row, new_col);
    }

    /// Erase from cursor to end of line.
    pub fn erase_to_end_of_line(&mut self) -> Result<()> {
        let row = self.cursor.position.row;
        let start_col = self.cursor.position.col;
        let cols = self.grid.cols();
        
        for col in start_col..cols {
            self.grid.write(row, col, Cell::default())?;
        }
        
        Ok(())
    }

    /// Erase fromstart of line to cursor.
    pub fn erase_to_start_of_line(&mut self) -> Result<()> {
        let row = self.cursor.position.row;
        let end_col = self.cursor.position.col;
        
        for col in 0..=end_col {
            self.grid.write(row, col, Cell::default())?;
        }
        
        Ok(())
    }

    /// Erase entire line.
    pub fn erase_line(&mut self) -> Result<()> {
        self.grid.clear_row(self.cursor.position.row)
    }

    /// Erase from cursor to end of screen.
    pub fn erase_to_end_of_screen(&mut self) -> Result<()> {
        // Erase current line from cursor
        self.erase_to_end_of_line()?;
        
        // Erase remaining lines
        for row in (self.cursor.position.row + 1)..self.grid.rows() {
            self.grid.clear_row(row)?;
        }
        
        Ok(())
    }

    /// Erase from start of screen to cursor.
    pub fn erase_to_start_of_screen(&mut self) -> Result<()> {
        // Erase lines before cursor
        for row in 0..self.cursor.position.row {
            self.grid.clear_row(row)?;
        }
        
        // Erase current line to cursor
        self.erase_to_start_of_line()?;
        
        Ok(())
    }

    /// Erase entire screen.
    pub fn erase_screen(&mut self) {
        self.grid.clear();
    }

    /// Scroll up by n lines.
    pub fn scroll_up(&mut self, n: usize) {
        self.grid.scroll_up(n);
    }

    /// Scroll down by n lines.
    pub fn scroll_down(&mut self, n: usize) {
        self.grid.scroll_down(n);
    }

    /// Resize the terminal.
    pub fn resize(&mut self, rows: usize, cols: usize) -> Result<()> {
        self.grid.resize(GridSize { rows, cols })?;
        self.scroll_region = (0, rows);
        Ok(())
    }

    /// Save cursor (DECSC).
    pub fn save_cursor(&mut self) {
        self.decsc_cursor = Some(self.cursor.position);
    }

    /// Restore cursor (DECRC).
    pub fn restore_cursor(&mut self) {
        if let Some(pos) = self.decsc_cursor {
            self.cursor.position = pos;
        }
    }

    /// Switch to alternate buffer.
    pub fn use_alternate_buffer(&mut self, use_alternate: bool) -> Result<()> {
        if use_alternate {
            // Save current state
            self.alternate_grid = Some(std::mem::replace(
                &mut self.grid,
                Grid::new(GridSize {
                    rows: self.grid.rows(),
                    cols: self.grid.cols(),
                }),
            ));
            self.alternate_cursor = Some(self.cursor.clone());
        } else if let Some(alt_grid) = self.alternate_grid.take() {
            // Restore normal buffer
            self.grid = alt_grid;
            if let Some(alt_cursor) = self.alternate_cursor.take() {
                self.cursor = alt_cursor;
            }
        }
        Ok(())
    }

    /// Get damaged regions.
    pub fn take_damage(&mut self) -> Vec<crate::damage::DamageRegion> {
        self.grid.take_damage()
    }
}

## Notes

### Design Decisions

1. **Grid as Vec<Vec<Cell>>**: Row-major order for cache locality. Alternative would be single Vec<Cell> with index calculation.

2. **Damage tracking**: Enables efficient partial redraws. Renderer only needs to process damaged regions.

3. **PtyBackend trait**: Platform abstraction from day one. Allows testing with mock PTY.

4. **No unsafe code**: Use safe Rust everywhere. If unsafe is needed,document with SAFTY comment.

5. **Error types**: Custom error enum with thiserror. No panics in production code.

### Future Considerations

1. **Grid alternative**: Contiguous Vec<Cell> may be faster for some operations.

2. **Damage optimization**: Merge adjacent cells into rectangles.

3. **Unicode width**: Update unicode-width crate regularly for emoji width changes.

4. **Thread safety**: Terminal struct is not thread-safe. Use channels for IPC if needed.
