## SIMD Parser Optimization

```rust
use std::arch::x86_64::*;

#[cfg(target_arch = "x86_64")]
pub fn find_esc_simd(data: &[u8]) -> Option<usize> {
    unsafe {
        let esc = _mm256_set1_epi8(0x1B);
        
        for (i, chunk) in data.chunks(32).enumerate() {
            let mask = if chunk.len() == 32 {
                let v = _mm256_loadu_si256(chunk.as_ptr() as *const __m256i);
                _mm256_cmpeq_epi8(v, esc)
            } else {
                let mut arr = [0u8; 32];
                arr[..chunk.len()].copy_from_slice(chunk);
                let v = _mm256_loadu_si256(arr.as_ptr() as *const __m256i);
                _mm256_cmpeq_epi8(v, esc)
            };
            
            let bitmask = _mm256_movemask_epi8(mask);
            if bitmask != 0 {
                let bit_pos = bitmask.trailing_zeros() as usize;
                return Some(i * 32 + bit_pos);
            }
        }
        None
    }
}

#[cfg(not(target_arch = "x86_64"))]
pub fn find_esc_simd(data: &[u8]) -> Option<usize> {
    data.iter().position(|&b| b == 0x1B)
}
```

## Grid Optimization

```rust
pub struct OptimizedGrid {
    /// Cells in row-major order.
    cells: Vec<Cell>,
    /// Attribute deduplication table.
    attrs_table: Arc<AttrsTable>,
    /// Number of columns.
    cols: usize,
    /// Number of rows.
    rows: usize,
}

pub struct AttrsTable {
    /// Unique attribute sets.
    attrs: Vec<CellAttrs>,
    /// Lookup from attrs to ID.
    lookup: HashMap<CellAttrs, u16>,
}

impl AttrsTable {
    pub fn get_or_insert(&mut self, attrs: CellAttrs) -> u16 {
        if let Some(&id) = self.lookup.get(&attrs) {
            return id;
        }
        let id = self.attrs.len() as u16;
        self.attrs.push(attrs.clone());
        self.lookup.insert(attrs, id);
        id
    }
}

impl OptimizedGrid {
    /// Write cell with deduplicated attrs.
    pub fn write(&mut self, row: usize, col: usize, char: char, attrs: CellAttrs) {
        let attrs_id = self.attrs_table.get_or_insert(attrs);
        let idx = row * self.cols + col;
        self.cells[idx] = Cell { char, width: CellWidth::from_char(char), attrs_id };
    }
}
```

## Zero-Copy PTY

```rust
pub struct ZeroCopyPty {
    buffer: [u8; 65536],
    len: usize,
}

impl ZeroCopyPty {
    /// Read directly into parseable buffer.
    pub fn read_and_parse(&mut self, parser: &mut Parser) -> Result<Vec<Action>> {
        let n = self.pty.read(&mut self.buffer[self.len..])?;
        self.len += n;
        
        // Parse without copying
        let actions = parser.parse(&self.buffer[..self.len]);
        
        // Preserve partial sequence
        let consumed = parser.consumed();
        self.buffer.copy_within(consumed..self.len, 0);
        self.len -= consumed;
        
        Ok(actions)
    }
}
```

#[bench]
fn bench_find_esc_scalar(b: &mut Bencher) {

#[bench]
fn bench_input_latency(b: &mut Bencher) {
    let mut terminal = Terminal::new(80, 24);
    let mut parser = Parser::new();
    
    b.iter(|| {
        // Simulate keypress
        let input = b"a\x1b[1mb\x1b[0mc\n";
        let actions = parser.parse(input);
        for action in actions {
            terminal.handle(action);
        }
        terminal.take_damage();
    });
}
