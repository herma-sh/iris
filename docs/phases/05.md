## API Contracts

### pty/windows.rs

```rust
use super::traits::PtyBackend;
use crate::error::{Error, Result};
use std::io::{Read, Write};
use windows::Win32::Foundation::*;
use windows::Win32::System::Console::*;

pub struct ConPty {
    handle: HANDLE,
    process: HANDLE,
    input_pipe: HANDLE,
    output_pipe: HANDLE,
}

impl ConPty {
    pub fn spawn(command: &str, args: &[&str], cols: u16, rows: u16) -> Result<Self> {
        unsafe {
            // Create pipes
            let mut input_read = HANDLE::default();
            let mut input_write = HANDLE::default();
            let mut output_read = HANDLE::default();
            let mut output_write = HANDLE::default();
            
            CreatePipe(&mut input_read, &mut input_write, None, 0);
            CreatePipe(&mut output_read, &mut output_write, None, 0);
            
            // Create pseudo console
            let size = COORD { X: cols as i16, Y: rows as i16 };
            let mut pcon = HANDLE::default();
            
            CreatePseudoConsole(
                size,
                input_read,
                output_write,
                0,
                &mut pcon,
            ).map_err(|e| Error::PtyError(e.to_string()))?;
            
            // Setup process
            let mut si = STARTUPINFOEXW::default();
            si.StartupInfo.cb = std::mem::size_of::<STARTUPINFOEXW>() as u32;
            
            let attr_list_size = 0;
            InitializeProcThreadAttributeList(
                None,
                1,
                0,
                &attr_list_size as *const _ as *mut _,
            );
            
            let mut attr_list = vec![0u8; attr_list_size as usize];
            si.lpAttributeList = attr_list.as_mut_ptr() as _;
            
            UpdateProcThreadAttribute(
                si.lpAttributeList,
                0,
                PROC_THREAD_ATTRIBUTE_PSEUDOCONSOLE,
                Some(&pcon as *const _ as *const _),
                std::mem::size_of::<HANDLE>(),
                None,
                None,
            );
            
            // Build command line
            let cmd = format!("{} {}", command, args.join(" "));
            let mut cmd_wide: Vec<u16> = cmd.encode_utf16().chain(std::iter::once(0)).collect();
            
            let mut pi = PROCESS_INFORMATION::default();
            
            CreateProcessW(
                None,
                PWSTR(cmd_wide.as_mut_ptr()),
                None,
                None,
                false,
                EXTENDED_STARTUPINFO_PRESENT,
                None,
                None,
                &si.StartupInfo,
                &mut pi,
            ).map_err(|e| Error::PtyError(e.to_string()))?;
            
            Ok(Self {
                handle: pcon,
                process: pi.hProcess,
                input_pipe: input_write,
                output_pipe: output_read,
            })
        }
    }
}

impl PtyBackend for ConPty {
    fn read(&mut self, buf: &mut [u8]) -> Result<usize> {
        unsafe {
            let mut bytes_read = 0;
            ReadFile(self.output_pipe, Some(buf), Some(&mut bytes_read), None)?;
            Ok(bytes_read as usize)
        }
    }
    
    fn write(&mut self, data: &[u8]) -> Result<usize> {
        unsafe {
            let mut bytes_written = 0;
            WriteFile(self.input_pipe, Some(data), Some(&mut bytes_written), None)?;
            Ok(bytes_written as usize)
        }
    }
    
    fn resize(&mut self, cols: u16, rows: u16) -> Result<()> {
        unsafe {
            let size = COORD { X: cols as i16, Y: rows as i16 };
            ResizePseudoConsole(self.handle, size);
            Ok(())
        }
    }
    
    fn is_alive(&self) -> bool {
        unsafe {
            let mut exit_code = 0;
            GetExitCodeProcess(self.handle, &mut exit_code);
            exit_code == STILL_ACTIVE.0
        }
    }
}
```

## High DPI Support

```rust
pub trait HighDpi {
    fn scale_factor(&self) -> f32;
    fn scale_font(&self, base_size: f32) -> f32 {
        base_size * self.scale_factor()
    }
}

#[cfg(target_os = "windows")]
impl HighDpi for Window {
    fn scale_factor(&self) -> f32 {
        unsafe {
            let dpi = GetDpiForWindow(self.hwnd());
            dpi as f32 / 96.0
        }
    }
}
```

```rust
pub trait ImeHandler {

    fn start_composition(&mut self);
    fn update_composition(&mut self, text: &str, cursor: usize);
    fn end_composition(&mut self);

#[cfg(target_os = "windows")]
pub struct WindowsIme {
    hwnd: HWND,
    context: HIMC,
}

impl ImeHandler for WindowsIme {
    fn set_position(&mut self, x: i32, y: i32) {
        unsafe {
            let mut cf = COMPOSITIONFORM::default();
            cf.dwStyle = CFS_POINT;
            cf.ptCurrentPos = POINT { x, y };
            ImmSetCompositionWindow(self.context, &cf);
        }
    }
    
    fn start_composition(&mut self) {
        unsafe {
            ImmNotifyIME(self.context, NI_COMPOSITIONSTR, CPS_START, 0);
        }
    }

        // Update composition string
    }
    
    fn end_composition(&mut self) {
        unsafe {
            ImmNotifyIME(self.context, NI_COMPOSITIONSTR, CPS_FINISH, 0);
        }
    }
    
    fn cancel_composition(&mut self) {
        unsafe {
            ImmNotifyIME(self.context, NI_COMPOSITIONSTR, CPS_CANCEL, 0);
        }
    }
```
