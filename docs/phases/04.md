# Phase 4: Scrollback & Search

**Duration**: 1-2 weeks  
**Priority**: P0 High  
**Dependencies**: Phase 3

## Overview

This phase implements the scrollback buffer (history of terminal output) and search functionality. The scrollback allows users to view previous output, while search enables finding specific content. Key challenges: memory management, efficient search, and performance with large histories.

use std::collections::VecDeque;

/// A line in the scrollback buffer.
#[derive(Clone, Debug)]
pub struct Line {
    /// Cells in the line.
    pub cells: Vec<crate::cell::Cell>,
    /// Whether this is a wrapped line.
    pub wrapped: bool,
    /// Timestamp when line was added.
    pub timestamp: std::time::Instant,
    /// Line number (for search results).
    pub number: u64,
}

impl Line {
    /// Create a new line.
    pub fn new(cells: Vec<crate::cell::Cell>, wrapped: bool) -> Self {
        Self {
            cells,
            wrapped,
            timestamp: std::time::Instant::now(),
            number: 0,
        }
    }

    /// Get the text content of the line.
    pub fn text(&self) -> String {
        self.cells.iter().map(|c| c.char).collect()
    }

    /// Check if line contains a pattern.
    pub fn contains(&self, pattern: &str, case_sensitive: bool) -> bool {
        let text = self.text();
        if case_sensitive {
            text.contains(pattern)
        } else {
            text.to_lowercase().contains(&pattern.to_lowercase())
        }
    }

    /// Get display width (accounting for wide characters).
    pub fn display_width(&self) -> usize {
        self.cells.iter().map(|c| c.width.columns()).sum()
    }
}

/// Scrollback buffer configuration.
#[derive(Clone, Debug)]
pub struct ScrollbackConfig {
    /// Maximum number of lines to keep.
    pub max_lines: usize,
    /// Maximum memory usage in bytes (approximate).
    pub max_memory: Option<usize>,
}

impl Default for ScrollbackConfig {
    fn default() -> Self {
        Self {
            max_lines: 10000,
            max_memory: None,
        }
    }
}

/// Scrollback buffer (ring buffer).
pub struct Scrollback {
    /// Configuration.
    config: ScrollbackConfig,
    /// Lines in the buffer.
    lines: VecDeque<Line>,
    /// Total lines ever seen (for line numbering).
    total_lines: u64,
    /// Current estimated memory usage.
    memory_usage: usize,
}

impl Scrollback {
    /// Create a new scrollback buffer.
    pub fn new(config: ScrollbackConfig) -> Self {
        Self {
            config,
            lines: VecDeque::new(),
            total_lines: 0,
            memory_usage: 0,
        }
    }

    /// Add a line to the buffer.
    pub fn push(&mut self, line: Line) {
        // Set line number
        let mut line = line;
        line.number = self.total_lines;
        self.total_lines += 1;
        
        // Estimate memory usage
        let line_size = line.cells.len() * std::mem::size_of::<crate::cell::Cell>();
        
        // Remove old lines if necessary
        while self.lines.len() >= self.config.max_lines {
            if let Some(old_line) = self.lines.pop_front() {
                self.memory_usage -= old_line.cells.len() * std::mem::size_of::<crate::cell::Cell>();
            }
        }
        
        // Check memory limit
        if let Some(max_memory) = self.config.max_memory {
            while self.memory_usage + line_size > max_memory && !self.lines.is_empty() {
                if let Some(old_line) = self.lines.pop_front() {
                    self.memory_usage -= old_line.cells.len() * std::mem::size_of::<crate::cell::Cell>();
                }
            }
        }
        
        // Add new line
        self.memory_usage += line_size;
        self.lines.push_back(line);
    }

    /// Get number of lines in buffer.
    pub fn len(&self) -> usize {
        self.lines.len()
    }

    /// Check if buffer is empty.
    pub fn is_empty(&self) -> bool {
        self.lines.is_empty()
    }

    /// Get a line byindex (0 = most recent).
    pub fn get(&self, index: usize) -> Option<&Line> {
        // Index from end
        let len = self.lines.len();
        if index < len {
            self.lines.get(len -1 - index)
        } else {
            None
        }
    }

    /// Get a line by line number.
    pub fn get_by_number(&self, number: u64) -> Option<&Line> {
        self.lines.iter().find(|l| l.number == number)
    }

    /// Iterate over lines (newest first).
    pub fn iter(&self) -> impl Iterator<Item = &Line> {
        self.lines.iter().rev()
    }

    /// Iterate over lines (oldest first).
    pub fn iter_oldest_first(&self) -> impl Iterator<Item = &Line> {
        self.lines.iter()
    }

    /// Search for lines matching a pattern.
    pub fn search(&self, pattern: &str, case_sensitive: bool) -> Vec<SearchResult> {
        if pattern.is_empty() {
            return Vec::new();
        }
        
        let mut results = Vec::new();
        
        for (i, line) in self.lines.iter().enumerate().rev() {
            if line.contains(pattern, case_sensitive) {
                // Find all occurrences in the line
                let text = line.text();
                let search_text = if case_sensitive {
                    text.as_str()
                } else {
                    &text.to_lowercase()
                };
                let search_pattern = if case_sensitive {
                    pattern
                } else {
                    &pattern.to_lowercase()
                };
                
                let mut start = 0;
                while let Some(pos) = search_text[start..].find(search_pattern) {
                    let abs_pos = start + pos;
                    results.push(SearchResult {
                        line_number: line.number,
                        line_index: i,
                        column: abs_pos,
                        length: pattern.len(),
                        text: line.text(),
                    });
                    start = abs_pos + search_pattern.len();
                    if start >= text.len() {
                        break;
                    }
                }
            }
        }
        
        results
    }

    /// Clear the buffer.
    pub fn clear(&mut self) {
        self.lines.clear();
        self.memory_usage = 0;
    }

    /// Get memory usage.
    pub fn memory_usage(&self) -> usize {
        self.memory_usage
    }
}

/// Search result.
#[derive(Clone, Debug)]
pub struct SearchResult {
    /// Line number.
    pub line_number: u64,
    /// Index in scrollback buffer.
    pub line_index: usize,
    /// Column of match start.
    pub column: usize,
    /// Length of match.
    pub length: usize,
    /// Full line text.
    pub text: String,
}

impl SearchResult {
    /// Get end column.
    pub fn end_column(&self) -> usize {
        self.column + self.length
    }
}

use regex::Regex;

/// Search configuration.
#[derive(Clone, Debug)]
pub struct SearchConfig {
    /// Search pattern.
    pub pattern: String,
    /// Case sensitive search.
    pub case_sensitive: bool,
    /// Use regex.
    pub regex: bool,
    /// Whole word search.
    pub whole_word: bool,
    /// Wrap around.
    pub wrap: bool,
}

impl Default for SearchConfig {
    fn default() -> Self {
        Self {
            pattern: String::new(),
            case_sensitive: false,
            regex: false,
            whole_word: false,
            wrap: true,
        }
    }
}

/// Search engine.
pub struct SearchEngine {
    /// Current search configuration.
    config: SearchConfig,
    /// Compiled regex (if using regex).
    compiled_regex: Option<Regex>,
    /// Current match index.
    current_match: usize,
}

impl SearchEngine {
    /// Create a new search engine.
    pub fn new() -> Self {
        Self {
            config: SearchConfig::default(),
            compiled_regex: None,
            current_match: 0,
        }
    }

    /// Set search pattern.
    pub fn set_pattern(&mut self, pattern: &str) {
        self.config.pattern = pattern.to_string();
        self.current_match = 0;
        
        // Compile regex if needed
        if self.config.regex {
            let flags = if self.config.case_sensitive {
                ""
            } else {
                "(?i)"
            };
            let pattern = format!("{}{}", flags, pattern);
            self.compiled_regex = Regex::new(&pattern).ok();
        } else {
            self.compiled_regex = None;
        }
    }

    /// Set case sensitivity.
    pub fn set_case_sensitive(&mut self, case_sensitive: bool) {
        self.config.case_sensitive = case_sensitive;
        // Recompile regex if using regex
        if self.config.regex {
            self.set_pattern(&self.config.pattern.clone());
        }
    }

    /// Search forward in scrollback.
    pub fn search_forward(
        &mut self,
        scrollback: &super::Scrollback,
        start_line: u64,
        start_col: usize,
    ) -> Option<super::SearchResult> {
        let results = self.search(scrollback)?;
        
        // Find next match after start position
        for (i, result) in results.iter().enumerate() {
            if result.line_number < start_line {
                continue;
            }
            if result.line_number == start_line && result.column <= start_col {
                continue;
            }
            self.current_match = i;
            return Some(result.clone());
        }
        
        // Wrap around if configured
        if self.config.wrap &&!results.is_empty() {
            self.current_match = 0;
            return results.first().cloned();
        }
        
        None
    }

    /// Search backward in scrollback.
    pub fn search_backward(
        &mut self,
        scrollback: &super::Scrollback,
        start_line: u64,
        start_col: usize,
    ) -> Option<super::SearchResult> {
        let results = self.search(scrollback)?;
        
        // Find previous match before start position
        for (i, result) in results.iter().enumerate().rev() {
            if result.line_number > start_line {
                continue;
            }
            if result.line_number == start_line && result.column >= start_col {
                continue;
            }
            self.current_match = i;
            return Some(result.clone());
        }
        
        // Wrap around if configured
        if self.config.wrap && !results.is_empty() {
            self.current_match = results.len() - 1;
            return results.last().cloned();
        }
        
        None
    }

    /// Get all matches.
    fn search(&self, scrollback: &super::Scrollback) -> Option<Vec<super::SearchResult>> {
        if self.config.pattern.is_empty() {
            return None;
        }
        
        if self.config.regex {
            self.search_regex(scrollback)
        } else if self.config.whole_word {
            self.search_whole_word(scrollback)
        } else {
            Some(scrollback.search(&self.config.pattern, self.config.case_sensitive))
        }
    }

    /// Search with regex.
    fn search_regex(&self, scrollback: &super::Scrollback) -> Option<Vec<super::SearchResult>> {
        let regex = self.compiled_regex.as_ref()?;
        let mut results = Vec::new();
        
        for line in scrollback.iter() {
            let text = line.text();
            for cap in regex.find_iter(&text) {
                results.push(super::SearchResult {
                    line_number: line.number,
                    line_index: 0, // Set by caller
                    column: cap.start(),
                    length: cap.end() - cap.start(),
                    text: text.clone(),
                });
            }
        }
        
        Some(results)
    }

    /// Search for whole words.
    fn search_whole_word(&self, scrollback: &super::Scrollback) -> Option<Vec<super::SearchResult>> {
        let mut results = Vec::new();
        
        for line in scrollback.iter() {
            let text = line.text();
            let search_text = if self.config.case_sensitive {
                text.clone()
            } else {
                text.to_lowercase()
            };
            let pattern = if self.config.case_sensitive {
                self.config.pattern.clone()
            } else {
                self.config.pattern.to_lowercase()
            };
            
            let mut start = 0;
            while let Some(pos) = search_text[start..].find(&pattern) {
                let abs_pos = start + pos;
                let end = abs_pos + pattern.len();
                
                // Check word boundaries
                let is_word_start = abs_pos == 0
                    || !search_text.as_bytes()[abs_pos - 1].is_ascii_alphanumeric();
                let is_word_end = end >= search_text.len()
                    || !search_text.as_bytes()[end].is_ascii_alphanumeric();
                
                if is_word_start && is_word_end {
                    results.push(super::SearchResult {
                        line_number: line.number,
                        line_index: 0,
                        column: abs_pos,
                        length: pattern.len(),
                        text: text.clone(),
                    });
                }
                
                start = abs_pos + pattern.len();
                if start >= search_text.len() {
                    break;
                }
            }
        }
        
        Some(results)
    }

    /// Get current match index.
    pub fn current_match(&self) -> usize {
        self.current_match
    }
}

impl Default for SearchEngine {
    fn default() -> Self {
        Self::new()
    }
}
