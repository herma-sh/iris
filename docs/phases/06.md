## API Contracts

### main.rs

```rust
mod cli;
mod config;
mod window;

use clap::Parser;
use iris_core::Terminal;
use iris_render_wgpu::Renderer;

fn main() -> anyhow::Result<()> {
    let args = cli::Args::parse();
    let config = config::load(&args.config)?;
    
    // Create window
    let event_loop = winit::event_loop::EventLoop::new();
    let window = window::create(&event_loop, &config)?;
    
    // Create terminal
    let pty = create_pty(&args, &config)?;
    let mut terminal = Terminal::new(config.rows, config.cols);
    
    // Create renderer
    let mut renderer = Renderer::new(&window, config.theme)?;
    
    // Event loop
    event_loop.run(move |event, _, control_flow| {
        match event {
            winit::event::Event::WindowEvent { event, .. } => {
                match event {
                    winit::event::WindowEvent::CloseRequested => {
                        *control_flow = winit::event_loop::ControlFlow::Exit;
                    }
                    winit::event::WindowEvent::Resized(size) => {
                        renderer.resize(size.width, size.height);
                    }
                    winit::event::WindowEvent::KeyboardInput { input, .. } => {
                        handle_keyboard_input(input, &pty);
                    }
                    winit::event::WindowEvent::ReceivedCharacter(c) => {
                        pty.write(&[c as u8]);
                    }
                    _ => {}
                }
            }
            winit::event::Event::MainEventsCleared => {
                // Read from PTY
                let mut buf = [0u8; 4096];
                if let Ok(n) = pty.read(&mut buf) {
                    let actions = parser.parse(&buf[..n]);
                    for action in actions {
                        terminal.handle(action);
                    }
                }
                
                // Render
                renderer.render(&terminal, terminal.take_damage());
            }
            _ => {}
        }
    });
}
```

### cli.rs

```rust
use clap::{Parser, Subcommand};

#[derive(Parser)]
#[command(name = "iris")]
#[command(about = "A modern terminal emulator")]
#[command(version)]
pub struct Args {
    /// Command to execute
    #[arg(short = 'e', long = "command")]
    pub command: Option<String>,
    
    /// Working directory
    #[arg(short = 'w', long = "working-directory")]
    pub working_directory: Option<String>,
    
    /// Configuration file path
    #[arg(short = 'c', long = "config")]
    pub config: Option<String>,
    
    /// Theme name or path
    #[arg(long = "theme")]
    pub theme: Option<String>,
    
    /// SSH connection
    #[arg(long = "ssh")]
    pub ssh: Option<String>,
    
    /// Serial port
    #[arg(long = "serial")]
    pub serial: Option<String>,
    
    /// Serial baud rate
    #[arg(long = "baud", default_value = "115200")]
    pub baud: u32,
    
    #[command(subcommand)]
    pub subcommand: Option<Commands>,
}

#[derive(Subcommand)]
pub enum Commands {
    /// List available themes
    Themes,
    /// List available fonts
    Fonts,
    /// Show configuration
    Config,
}
```

### config.rs

```rust
use serde::Deserialize;
use std::path::Path;

#[derive(Clone, Debug, Deserialize)]
pub struct Config {
    pub terminal: TerminalConfig,
    pub font: FontConfig,
    pub window: WindowConfig,
    pub theme: ThemeConfig,
}

#[derive(Clone, Debug, Deserialize)]
pub struct TerminalConfig {
    #[serde(default = "default_shell")]
    pub shell: String,
    
    #[serde(default = "default_rows")]
    pub rows: usize,
    
    #[serde(default = "default_cols")]
    pub cols: usize,
    
    #[serde(default = "default_scrollback")]
    pub scrollback_lines: usize,
}

#[derive(Clone, Debug, Deserialize)]
pub struct FontConfig {
    #[serde(default = "default_font_family")]
    pub family: String,
    
    #[serde(default = "default_font_size")]
    pub size: f32,
    
    #[serde(default = "default_line_height")]
    pub line_height: f32,
}

fn default_shell() -> String {
    std::env::var("SHELL").unwrap_or_else(|_| "bash".to_string())
}

fn default_rows() -> usize { 24 }
fn default_cols() -> usize { 80 }
fn default_scrollback() -> usize { 10000 }
fn default_font_family() -> String { "JetBrains Mono".to_string() }
fn default_font_size() -> f32 { 14.0 }
fn default_line_height() -> f32 { 1.2 }

impl Config {
    pub fn load(path: &Path) -> anyhow::Result<Self> {
        let content = std::fs::read_to_string(path)?;
        let config: Config = toml::from_str(&content)?;
        Ok(config)
    }
    
    pub fn default_path() -> std::path::PathBuf {
        dirs::config_local_dir()
            .unwrap_or_else(|| std::path::PathBuf::from("."))
            .join("iris")
            .join("config.toml")
    }
}
```
