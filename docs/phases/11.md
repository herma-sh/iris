### inspector/mod.rs

```rust
use iris_core::cell::{Cell, CellAttrs};
use iris_core::cursor::CursorPosition;

/// Cell information for display.
#[derive(Clone, Debug)]
pub struct CellInfo {
    /// Position in grid.
    pub position: CursorPosition,
    /// Character.
    pub character: char,
    /// Unicode codepoint.
    pub codepoint: u32,
    /// Character width (1 or 2).
    pub width: u8,
    /// Cell attributes.
    pub attrs: CellAttrs,
    /// Grid index.
    pub grid_index: usize,
    /// Damage state.
    pub is_damaged: bool,
}

/// Inspector state.
pub struct Inspector {
    /// Whether inspector is enabled.
    pub enabled: bool,
    /// Current cell under inspection.
    pub cell_info: Option<CellInfo>,
    /// Show color palette.
    pub show_palette: bool,
    /// Show protocol monitor.
    pub show_protocol: bool,
    /// Show performance metrics.
    pub show_metrics: bool,
}

impl Inspector {
    pub fn new() -> Self {
        Self {
            enabled: false,
            cell_info: None,
            show_palette: false,
            show_protocol: false,
            show_metrics: false,
        }
    }
    
    /// Inspect cell at position.
    pub fn inspect(&mut self, grid: &iris_core::grid::Grid, pos: CursorPosition) {
        if let Some(cell) = grid.cell(pos.row, pos.col) {
            let attrs = cell.attrs.clone();
            let grid_index = pos.row * grid.cols() + pos.col;
            
            self.cell_info = Some(CellInfo {
                position: pos,
                character: cell.char,
                codepoint: cell.char as u32,
                width: cell.width.columns() as u8,
                attrs,
                grid_index,
                is_damaged: grid.is_cell_damaged(pos.row, pos.col),
            });
        }
    }
    
    /// Clear inspection.
    pub fn clear(&mut self) {
        self.cell_info = None;
    }
    
    /// Toggle inspector.
    pub fn toggle(&mut self) {
        self.enabled = !self.enabled;
    }
}
```

### inspector/metrics.rs

```rust
use std::time::{Duration, Instant};

/// Performance metrics.
#[derive(Clone, Debug, Default)]
pub struct Metrics {
    /// Frames per second.
    pub fps: f64,
    /// Frame time in ms.
    pub frame_time_ms: f64,
    /// Input latency in ms.
    pub input_latency_ms: f64,
    /// Memory usage in bytes.
    pub memory_bytes: usize,
    /// Number of cells.
    pub cell_count: usize,
    /// Scrollback lines.
    pub scrollback_lines: usize,
    /// Parse throughput in MB/s.
    pub parse_throughput_mbs: f64,
}

pub struct MetricsTracker {
    frame_times: Vec<Instant>,
    input_latencies: Vec<Duration>,
}

impl MetricsTracker {
    pub fn new() -> Self {
        Self {
            frame_times: Vec::with_capacity(60),
            input_latencies: Vec::with_capacity(100),
        }
    }
    
    pub fn record_frame(&mut self) {
        self.frame_times.push(Instant::now());
        if self.frame_times.len() >60 {
            self.frame_times.remove(0);
        }
    }
    
    pub fn record_input_latency(&mut self, latency: Duration) {
        self.input_latencies.push(latency);
        if self.input_latencies.len() > 100 {
            self.input_latencies.remove(0);
        }
    }
    
    pub fn fps(&self) -> f64 {
        if self.frame_times.len() < 2 {
            return 0.0;
        }
        let duration = self.frame_times.last()
            .and_then(|end| {
                self.frame_times.first().map(|start| end.duration_since(*start))
            })
            .unwrap_or(Duration::ZERO);
        
        let frames = self.frame_times.len() - 1;
        frames as f64/ duration.as_secs_f64()
    }
    
    pub fn avg_input_latency(&self) -> Duration {
        if self.input_latencies.is_empty() {
            return Duration::ZERO;
        }
        self.input_latencies.iter().sum::<Duration>() /self.input_latencies.len() as u32
    }
    
    pub fn memory_usage(&self) -> usize {
        #[cfg(target_os = "linux")]
        {
            let mut stats = std::mem::MaybeUninit::uninit();
            unsafe {
                libc::getrusage(libc::RUSAGE_SELF, stats.as_mut_ptr());
                stats.assume_init().ru_maxrss as usize * 1024
            }
        }
        #[cfg(not(target_os = "linux"))]
        {
            0
        }
    }
    
    pub fn metrics(&self, grid: &iris_core::grid::Grid, scrollback_len: usize) -> Metrics {
        Metrics {
            fps: self.fps(),
            frame_time_ms: 1000.0 / self.fps().max(1.0),
            input_latency_ms: self.avg_input_latency().as_secs_f64() * 1000.0,
            memory_bytes: self.memory_usage(),
            cell_count: grid.rows() * grid.cols(),
            scrollback_lines: scrollback_len,
            parse_throughput_mbs: 0.0,  // Set by parser
        }
    }
}
```

### inspector/protocol_monitor.rs

```rust
use std::time::Instant;

/// Log entry for protocol monitor.
#[derive(Clone, Debug)]
pub struct LogEntry {
    pub timestamp: Instant,
    pub direction: Direction,
    pub sequence: Sequence,
}

#[derive(Clone, Copy, Debug)]
pub enum Direction {
    In,   // From PTY to terminal
    Out,// From terminal to PTY
}

#[derive(Clone, Debug)]
pub enum Sequence {
    Char(char),
    Ctrl(char),
    Escape(String),
    Csi(String),
    Osc(String),
    Dcs(String),
    Unknown(Vec<u8>),
}

pub struct ProtocolMonitor {
    entries: Vec<LogEntry>,
    max_entries: usize,
    filter: MonitorFilter,
}

#[derive(Clone, Debug, Default)]
pub struct MonitorFilter {
    pub show_in: bool,
    pub show_out: bool,
    pub show_control: bool,
    pub show_escape: bool,
}

impl ProtocolMonitor {
    pub fn new(max_entries: usize) -> Self {
        Self {
            entries: Vec::with_capacity(max_entries),
            max_entries,
            filter: MonitorFilter::default(),
        }
    }
    
    pub fn log(&mut self, direction: Direction, data: &[u8], parser: &mut Parser) {
        if !self.filter.matches(direction) {
            return;
        }
        
        for byte in data {
            if let Some(sequence) = parser.parse_byte(*byte).last() {
                if self.filter.matches_sequence(&sequence) {
                    self.entries.push(LogEntry {
                        timestamp: Instant::now(),
                        direction,
                        sequence: Sequence::from_action(&sequence),
                    });
                }
            }
        }
        
        if self.entries.len() > self.max_entries {
            self.entries.remove(0);
        }
    }
    
    pub fn entries(&self) -> &[LogEntry] {
        &self.entries
    }
    
    pub fn clear(&mut self) {
        self.entries.clear();
    }
}
```

### Color Palette Display

```rust
use iris_core::cell::Color;

pub struct ColorPalette {
    mode: PaletteMode,
}

#[derive(Clone, Copy, Debug)]
pub enum PaletteMode {
    Color16,
    Color256,
    TrueColor,
}

impl ColorPalette {
    pub fn new() -> Self {
        Self { mode: PaletteMode::Color16 }
    }
    
    pub fn colors_16(&self) -> Vec<Color> {
        vec![
            Color::Ansi(0),// Black
            Color::Ansi(1),  // Red
            Color::Ansi(2),  // Green
            Color::Ansi(3),  // Yellow
            Color::Ansi(4),  // Blue
            Color::Ansi(5),  // Magenta
            Color::Ansi(6),  // Cyan
            Color::Ansi(7),  // White
            Color::Ansi(8),  // Bright Black
            Color::Ansi(9),  // Bright Red
            Color::Ansi(10), // Bright Green
            Color::Ansi(11), // Bright Yellow
            Color::Ansi(12), // Bright Blue
            Color::Ansi(13), // Bright Magenta
            Color::Ansi(14), // Bright Cyan
            Color::Ansi(15), // Bright White
        ]
    }
    
    pub fn colors_256(&self) -> Vec<Color> {
        let mut colors = Vec::with_capacity(256);
        
        // 16 standard colors
        colors.extend(self.colors_16());
        
        // 216 color cube (6x6x6)
        for r in 0..6 {
            for g in 0..6 {
                for b in 0..6 {
                    let rv = if r ==0 { 0 } else { r * 40 + 55 };
                    let gv = if g == 0 { 0 } else { g * 40 + 55 };
                    let bv = if b == 0 { 0 } else { b * 40 + 55 };
                    colors.push(Color::Rgb { r: rv, g: gv, b: bv });
                }
            }
        }
        
        // 24 grayscale
        for i in 0..24 {
            let v = 8 + i * 10;
            colors.push(Color::Rgb { r: v, g: v, b: v });
        }
        
        colors
    }
}
```
