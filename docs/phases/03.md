```rust
use crate::cursor::CursorPosition;

/// Selection kind.
#[derive(Clone, Copy, Debug, Default, PartialEq, Eq)]
pub enum SelectionKind {
    #[default]
    /// Character-by-character.
    Simple,
    /// Word boundary.
    Word,
    /// Full line.
    Line,
    /// Rectangular block.
    Block,
}

/// Selection state.
#[derive(Clone, Copy, Debug, Default, PartialEq, Eq)]
pub enum SelectionState {
    #[default]
    /// No selection.
    None,
    /// Selecting (mouse down).
    Selecting,
    /// Selection complete (mouse up).
    Complete,
}

/// Selection anchor point.
#[derive(Clone, Copy, Debug, PartialEq, Eq)]
pub struct Anchor {
    /// Position in grid.
    pub position: CursorPosition,
    /// Kind of selection.
    pub kind: SelectionKind,
}

impl Anchor {
    pub fn new(row: usize, col: usize, kind: SelectionKind) -> Self {
        Self {
            position: CursorPosition::new(row, col),
            kind,
        }
    }
}

/// Selection range.
#[derive(Clone, Debug, PartialEq, Eq)]
pub struct Selection {
    /// Start anchor.
    pub start: Anchor,
    /// End anchor (cursor position during selection).
    pub end: Anchor,
    /// Selection kind.
    pub kind: SelectionKind,
    /// Selection state.
    pub state: SelectionState,
}

impl Selection {
    /// Create a new selection.
    pub fn new(start_row: usize, start_col: usize, kind: SelectionKind) -> Self {
        let start = Anchor::new(start_row, start_col, kind);
        let end = start.clone();
        Self {
            start,
            end,
            kind,
            state: SelectionState::Selecting,
        }
    }

    /// Check if selection is empty.
    pub fn is_empty(&self) -> bool {
        self.start.position == self.end.position
    }

    /// Check if a position is within selection.
    pub fn contains(&self, row: usize, col: usize) -> bool {
        match self.kind {
            SelectionKind::Simple | SelectionKind::Word | SelectionKind::Line => {
                self.contains_linear(row, col)
            }
            SelectionKind::Block => self.contains_block(row, col),
        }
    }

    /// Linear containment check.
    fn contains_linear(&self, row: usize, col: usize) -> bool {
        let (start, end) = self.ordered();
        
        if row < start.position.row || row > end.position.row {
            return false;
        }
        
        if row == start.position.row && row == end.position.row {
            return col >= start.position.col && col <= end.position.col;
        }
        
        if row == start.position.row {
            return col >= start.position.col;
        }
        
        if row == end.position.row {
            return col <= end.position.col;
        }
        
        true
    }

    /// Block containment check.
    fn contains_block(&self, row: usize, col: usize) -> bool {
        let min_row = self.start.position.row.min(self.end.position.row);
        let max_row = self.start.position.row.max(self.end.position.row);
        let min_col = self.start.position.col.min(self.end.position.col);
        let max_col = self.start.position.col.max(self.end.position.col);
        
        row >= min_row && row <= max_row && col >= min_col && col <= max_col
    }

    /// Get ordered (top-left, bottom-right) anchors.
    fn ordered(&self) -> (&Anchor, &Anchor) {
        if self.start.position.row < self.end.position.row {
            return (&self.start, &self.end);
        }
        if self.start.position.row > self.end.position.row {
            return (&self.end, &self.start);
        }
        if self.start.position.col <= self.end.position.col {
            (&self.start, &self.end)
        } else {
            (&self.end, &self.start)
        }
    }

    /// Extend selection to a new position.
    pub fn extend(&mut self, row: usize, col: usize) {
        self.end.position.row = row;
        self.end.position.col = col;
    }

    /// Get all selected positions.
    pub fn positions(&self) -> Vec<CursorPosition> {
        match self.kind {
            SelectionKind::Simple | SelectionKind::Word | SelectionKind::Line => {
                self.positions_linear()
            }
            SelectionKind::Block => self.positions_block(),
        }
    }

    /// Linear positions.
    fn positions_linear(&self) -> Vec<CursorPosition> {
        let (start, end) = self.ordered();
        let mut positions = Vec::new();
        
        for row in start.position.row..=end.position.row {
            let start_col = if row == start.position.row {
                start.position.col
            } else {
                0
            };
            let end_col = if row == end.position.row {
                end.position.col
            } else {
                usize::MAX
            };
            
            for col in start_col..=end_col {
                positions.push(CursorPosition::new(row, col));
            }
        }
        
        positions
    }

    /// Block positions.
    fn positions_block(&self) -> Vec<CursorPosition> {
        let min_row = self.start.position.row.min(self.end.position.row);
        let max_row = self.start.position.row.max(self.end.position.row);
        let min_col = self.start.position.col.min(self.end.position.col);
        let max_col = self.start.position.col.max(self.end.position.col);
        
        let mut positions = Vec::new();
        for row in min_row..=max_row {
            for col in min_col..=max_col {
                positions.push(CursorPosition::new(row, col));
            }
        }
        positions
    }
}
```

```rust
use super::types::{Anchor, Selection, SelectionKind, SelectionState};
use crate::grid::Grid;
use crate::cursor::CursorPosition;

/// Selection engine.
pub struct SelectionEngine {
    /// Current selection.
    selection: Option<Selection>,
    /// Word boundaries for word selection.
    word_boundaries: fn(char) -> bool,
}

impl SelectionEngine {
    /// Create a new selection engine.
    pub fn new() -> Self {
        Self {
            selection: None,
            word_boundaries: is_word_boundary,
        }
    }

    /// Get current selection.
    pub fn selection(&self) -> Option<&Selection> {
        self.selection.as_ref()
    }

    /// Start a new selection.
    pub fn start(&mut self, row: usize, col: usize, kind: SelectionKind) {
        self.selection = Some(Selection::new(row, col, kind));
    }

    /// Extend current selection.
    pub fn extend(&mut self, row: usize, col: usize) {
        if let Some(ref mut sel) = self.selection {
            sel.extend(row, col);
        }
    }

    /// Complete selection.
    pub fn complete(&mut self) {
        if let Some(ref mut sel) = self.selection {
            sel.state = SelectionState::Complete;
        }
    }

    /// Cancel selection.
    pub fn cancel(&mut self) {
        self.selection = None;
    }

    /// Check if there's an active selection.
    pub fn is_selecting(&self) -> bool {
        self.selection
            .as_ref()
            .map(|s| s.state == SelectionState::Selecting)
            .unwrap_or(false)
    }

    /// Check if there's a completed selection.
    pub fn has_selection(&self) -> bool {
        self.selection
            .as_ref()
            .map(|s| !s.is_empty() && s.state == SelectionState::Complete)
            .unwrap_or(false)
    }

    /// Select word at position.
    pub fn select_word(&mut self, grid: &Grid, row: usize, col: usize) {
        let (start, end) = self.find_word_bounds(grid, row, col);
        self.selection = Some(Selection::new(start.row, start.col, SelectionKind::Word));
        if let Some(ref mut sel) = self.selection {
            sel.extend(end.row, end.col);
            sel.state = SelectionState::Complete;
        }
    }

    /// Select line at position.
    pub fn select_line(&mut self, grid: &Grid, row: usize) {
        let cols = grid.cols();
        self.selection = Some(Selection::new(row, 0, SelectionKind::Line));
        if let Some(ref mut sel) = self.selection {
            sel.extend(row, cols - 1);
            sel.state = SelectionState::Complete;
        }
    }

    /// Find word boundaries.
    fn find_word_bounds(&self, grid: &Grid, row: usize, col: usize) -> (CursorPosition, CursorPosition) {
        let start_col = self.find_word_start(grid, row, col);
        let end_col = self.find_word_end(grid, row, col);
        
        (
            CursorPosition::new(row, start_col),
            CursorPosition::new(row, end_col),
        )
    }

    /// Find start of word.
    fn find_word_start(&self, grid: &Grid, row: usize, col: usize) -> usize {
        let row_cells = match grid.row(row) {
            Some(r) => r,
            None => return col,
        };
        
        let mut start_col = col;
        while start_col > 0 {
            let prev_char = row_cells[start_col - 1].char;
            if (self.word_boundaries)(prev_char) {
                break;
            }
            start_col -= 1;
        }
        start_col
    }

    /// Find end of word.
    fn find_word_end(&self, grid: &Grid, row: usize, col: usize) -> usize {
        let row_cells = match grid.row(row) {
            Some(r) => r,
            None => return col,
        };
        
        let mut end_col = col;
        while end_col < row_cells.len() - 1 {
            let next_char = row_cells[end_col + 1].char;
            if (self.word_boundaries)(next_char) {
                break;
            }
            end_col += 1;
        }
        end_col
    }

    /// Get selected text.
    pub fn get_selected_text(&self, grid: &Grid) -> Option<String> {
        let sel = self.selection.as_ref()?;
        let positions = sel.positions();
        
        let mut text = String::new();
        let mut prev_row = None;
        
        for pos in positions {
            if let Some(prev) = prev_row {
                if prev != pos.row {
                    text.push('\n');
                }
            }
            prev_row = Some(pos.row);
            
            if let Some(cell) = grid.cell(pos.row, pos.col) {
                text.push(cell.char);
            }
        }
        
        Some(text)
    }
}

/// Check if character is a word boundary.
fn is_word_boundary(c: char) -> bool {
    c.is_whitespace() || c.is_ascii_punctuation()
}

impl Default for SelectionEngine {
    fn default() -> Self {
        Self::new()
    }
}

mod tests {
```

```rust
#[cfg(target_os = "windows")]
mod windows {
    use super::Clipboard;
    use crate::error::{Error, Result};
    use windows::Win32::Foundation::*;
    use windows::Win32::System::DataExchange::*;

    pub struct WindowsClipboard;

    impl WindowsClipboard {
        fn open(&self) -> std::result::Result<(), windows::core::Error> {
            unsafe {
                OpenClipboard(None)?;
            }
            Ok(())
        }

        fn close(&self) {
            unsafe {
                let _ = CloseClipboard();
            }
        }
    }

    impl Clipboard for WindowsClipboard {
        fn get_text(&mut self) -> Result<String> {
            self.open().map_err(|e| Error::ClipboardError(e.to_string()))?;
            
            let text = unsafe {
                let handle = GetClipboardData(CF_UNICODETEXT.0);
                if handle.is_null() {
                    self.close();
                    return Err(Error::ClipboardError("GetClipboardData failed".into()));
                }
                
                let ptr = GlobalLock(handle) as *const u16;
                if ptr.is_null() {
                    self.close();
                    return Err(Error::ClipboardError("GlobalLock failed".into()));
                }
                
                let len = GlobalSize(handle) /2;
                let slice = std::slice::from_raw_parts(ptr, len);
                let text = String::from_utf16_lossy(slice);
                
                GlobalUnlock(handle);
                text
            };
            
            self.close();
            Ok(text.trim_end_matches('\0').to_string())
        }

        fn set_text(&mut self, text: &str) -> Result<()> {
            self.open().map_err(|e| Error::ClipboardError(e.to_string()))?;
            
            unsafe {
                EmptyClipboard();
                
                let wide_text = text.encode_utf16().chain(std::iter::once(0)).collect::<Vec<u16>>();
                let size = wide_text.len() * 2;
                
                let handle = GlobalAlloc(GMEM_FIXED, size);
                if handle.is_null() {
                    self.close();
                    return Err(Error::ClipboardError("GlobalAlloc failed".into()));
                }
                
                let ptr = GlobalLock(handle) as *mut u16;
                std::ptr::copy_nonoverlapping(wide_text.as_ptr(), ptr, wide_text.len());
                GlobalUnlock(handle);
                
                SetClipboardData(CF_UNICODETEXT.0, handle);
            }
            
            self.close();
            Ok(())
        }
    }
}
```

## Notes

### Key Design Decisions

1. **Selection engine is part of iris-core**: Platform-agnostic selection logic
2. **Clipboard is platform-specific**: iris-platform handles OS differences
3. **Block selection is rectangular**: Ignores line structure
4. **Word boundaries use Unicode**: `is_whitespace()`` and `is_ascii_punctuation()`

### Edge Cases

- Selection across wrapped lines
- Selection with wide characters (CJK, emoji)
- Selection in alternate screen buffer
- Selection during scrollback navigation
- Empty selection handling
