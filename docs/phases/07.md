## Overview

This phase creates the Tauri plugin for Iris, enabling it to render inside the Hermes Electron/WebView. Uses `raw-window-handle` for surface sharing.

### @iris/react/TerminalSurface.tsx

```tsx
import { useEffect, useRef, useState } from 'react';
import { invoke } from '@tauri-apps/api/tauri';
import { listen } from '@tauri-apps/api/event';
import { TerminalOptions, TerminalEvent, KeyEvent, MouseEvent } from '@iris/contract';

export interface TerminalSurfaceProps extends TerminalOptions {
  className?: string;
  onTitle?: (title: string) => void;
  onExit?: (code: number) => void;
  onBell?: () => void;
}

export function TerminalSurface({
  id,
  cols,
  rows,
  fontFamily,
  fontSize,
  theme,
  className,
  onTitle,
  onExit,
  onBell,
}: TerminalSurfaceProps) {
  const containerRef = useRef<HTMLDivElement>(null);
  const [ready, setReady] = useState(false);

  useEffect(() => {
    // Create terminal
    invoke('create_terminal', { id, cols, rows, fontFamily, fontSize, theme })
      .then(() => setReady(true))
      .catch(console.error);

    // Listen for events
    const unlistenData = listen<TerminalEvent>(`terminal:${id}`, (event) => {
      switch (event.payload.type) {
        case 'title':
          onTitle?.(event.payload.payload as string);
          break;
        case 'exit':
          onExit?.(event.payload.payload as number);
          break;
        case 'bell':
          onBell?.();
          break;
      }
    });

    return () => {
      unlistenData.then((f) => f());
      invoke('destroy_terminal', { id }).catch(console.error);
    };
  }, [id]);

  // Keyboard input
  const handleKeyDown = (e: React.KeyboardEvent) => {
    e.preventDefault();
    
    const keyEvent: KeyEvent = {
      key: e.key,
      code: e.code,
      ctrl: e.ctrlKey,
      alt: e.altKey,
      shift: e.shiftKey,
      meta: e.metaKey,
    };

    invoke('send_key_event', { id, event: keyEvent });
  };

  // Mouse input
  const handleMouseDown = (e: React.MouseEvent) => {
    const rect = containerRef.current?.getBoundingClientRect();
    if (!rect) return;

    const mouseEvent: MouseEvent = {
      type: 'press',
      x: e.clientX - rect.left,
      y: e.clientY - rect.top,
      button: e.button,
      ctrl: e.ctrlKey,
      alt: e.altKey,
      shift: e.shiftKey,
      meta: e.metaKey,
    };

    invoke('send_mouse_event', { id, event: mouseEvent });
  };

  return (
    <div
      ref={containerRef}
      className={className}
      onKeyDown={handleKeyDown}
      onMouseDown={handleMouseDown}
      tabIndex={0}
    >
      {/* Surface managed by Rust */}
    </div>
  );
}
```

use tauri::{command, App, Manager, Window};
use std::collections::HashMap;
use std::sync::{Arc, Mutex};

pub struct IrisState {
    terminals: HashMap<String, Arc<Mutex<TerminalHandle>>>,
}

pub struct TerminalHandle {
    terminal: iris_core::Terminal,
    pty: Box<dyn iris_platform::PtyBackend>,
}

#[command]
pub async fn create_terminal(
    id: String,
    cols: usize,
    rows: usize,
    font_family: Option<String>,
    font_size: Option<f32>,
    theme: Option<ThemeConfig>,
    window: Window,
    state: tauri::State<'_, IrisState>,
) -> Result<(), String> {
    // Create terminal
    let terminal = iris_core::Terminal::new(rows, cols);
    
    // Create PTY
    let pty = iris_platform::create_pty()
        .map_err(|e| e.to_string())?;
    
    let handle = Arc::new(Mutex::new(TerminalHandle { terminal, pty }));
    
    state.terminals.insert(id, handle);
    
    // Get window handle for surface
    let handle = window.window_handle()
        .map_err(|e| e.to_string())?;
    
    // Create renderer with handle
    // ...
    
    Ok(())
}

#[command]
pub async fn destroy_terminal(
    id: String,
    state: tauri::State<'_, IrisState>,
) -> Result<(), String> {
    state.terminals.remove(&id);
    Ok(())
}

#[command]
pub async fn write_to_terminal(
    id: String,
    data: String,
    state: tauri::State<'_, IrisState>,
) -> Result<(), String> {
    if let Some(handle) = state.terminals.get(&id) {
        let mut h = handle.lock().map_err(|e| e.to_string())?;
        h.pty.write(data.as_bytes())
            .map_err(|e| e.to_string())?;
    }
    Ok(())
}

#[command]
pub async fn send_key_event(
    id: String,
    event: KeyEvent,
    state: tauri::State<'_, IrisState>,
) -> Result<(), String> {
    if let Some(handle) = state.terminals.get(&id) {
        let mut h = handle.lock().map_err(|e| e.to_string())?;
        // Convert KeyEvent to terminal input
        let bytes = key_event_to_bytes(&event);
        h.pty.write(&bytes)
            .map_err(|e| e.to_string())?;
    }
    Ok(())
}

pub fn init() -> impl Fn(&mut App) {
    |app| {
        app.manage(IrisState {
            terminals: HashMap::new(),
        });
        Ok(())
    }
}
